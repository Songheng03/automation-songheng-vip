JavaScript

async function useMyAutomaton(text) {
  const res = await fetch('http://automation.songheng.vip:8888/v1/analyze', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text })
  });
  if (res.status === 402) {
    const { amount_usdc_cents, address } = await res.json();
    const txHash = await sendUSDC(address, amount_usdc_cents / 100);
    return fetch('http://automation.songheng.vip:8888/v1/analyze', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-X402-Payment': txHash
      },
      body: JSON.stringify({ text })
    }).then(r => r.json());
  }
  return res.json();
}
