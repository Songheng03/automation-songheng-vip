One-Click Setup

Add this workflow file to your repo manually:

name: AI Code Review
on:
  pull_request:
    types: [opened, synchronize]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
      - name: Get diff
        id: diff
        run: |
          DIFF=$(git diff origin/${{ github.base_ref }}...HEAD -- . || true)
          echo "diff<> $GITHUB_OUTPUT
          echo "$DIFF" >> $GITHUB_OUTPUT
          echo "EOF" >> $GITHUB_OUTPUT
      - name: AI Code Review
        id: review
        run: |
          RESPONSE=$(curl -s -X POST https://automation.songheng.vip/api/free/review \
            -H "Content-Type: application/json" \
            -d "{\"text\": \"${{ steps.diff.outputs.diff }}\"}")
          echo "result<> $GITHUB_OUTPUT
          echo "$RESPONSE" >> $GITHUB_OUTPUT
          echo "EOF" >> $GITHUB_OUTPUT
      - name: Post review comment
        uses: actions/github-script@v7
        with:
          script: |
            const review = JSON.parse(process.env.REVIEW_RESULT || '{}');
            const body = '## 🤖 AI Code Review\n\n' +
              (review.result || review.response || 'No issues found.') +
              '\n\n---\n_Powered by [my-automaton](https://automation.songheng.vip)_';
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: body
            });
        env:
          REVIEW_RESULT: ${{ steps.review.outputs.result }}

Save as .github/workflows/ai-code-review.yml in your repo.

API Reference

All endpoints accept POST requests with JSON body {"text": "your input"}.

# Free tier (3/day per service)
curl -X POST https://automation.songheng.vip/api/free/review \
  -H "Content-Type: application/json" \
  -d '{"text": "const x = 1 + 1"}'
# Premium tier (requires API key)
curl -X POST https://automation.songheng.vip/v1/review \
  -H "Content-Type: application/json" \
  -H "X-API-Key: am_YOUR_KEY" \
