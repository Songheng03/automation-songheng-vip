    runs-on: ubuntu-latest

      - uses: actions/checkout@v4
      - name: AI Code Review
        env:
          API_KEY: ${{ secrets.MY_AUTOMATON_KEY }}
        run: |
          # Review all changed JS/TS files
          for file in $(git diff --name-only origin/main | grep -E '\.(js|ts|py|java)$'); do
            echo "=== Reviewing: $file ==="
            curl -s -X POST https://automation.songheng.vip/v1/review \
              -H "Content-Type: application/json" \
              -H "X-API-Key: $API_KEY" \
              -d "{\"code\":$(cat $file | jq -Rs .),\"language\":\"${file##*.}\"}" | jq .
          done
📋 Copy#!/usr/bin/env python3
"""CI script: automatically review code in PRs."""
import os, sys, json, requests

API_KEY = os.environ.get('MY_AUTOMATON_KEY')
BASE = "https://automation.songheng.vip"

def review_file(filepath):
    ext_map = {'.py': 'python', '.js': 'javascript', '.ts': 'typescript',
                '.java': 'java', '.go': 'go', '.rs': 'rust'}
    ext = os.path.splitext(filepath)[1].lower()
    lang = ext_map.get(ext, 'text')
    
    with open(filepath) as f:
        code = f.read()
    
    resp = requests.post(f"{BASE}/v1/review", 
        json={"code": code, "language": lang},
        headers={"X-API-Key": API_KEY, "Content-Type": "application/json"})
    return resp.json()

# Review all files from git diff
changed = os.popen("git diff --name-only origin/main").read().strip().split()
for f in changed:
    if os.path.isfile(f):
        print(f"Reviewing {f}...")
        result = review_file(f)
        print(json.dumps(result, indent=2))
📋 Copy#!/usr/bin/env node
/** CI script to auto-review changed files */
const fs = require('fs');
const { execSync } = require('child_process');

const API_KEY = process.env.MY_AUTOMATON_KEY;
const BASE = 'https://automation.songheng.vip';

const EXT_MAP = { '.py': 'python', '.js': 'javascript', '.ts': 'typescript',
  '.java': 'java', '.go': 'go', '.rs': 'rust' };

async function reviewFile(filepath) {
  const ext = require('path').extname(filepath);
  const lang = EXT_MAP[ext] || 'text';
  const code = fs.readFileSync(filepath, 'utf-8');
  
  const res = await fetch(`${BASE}/v1/review`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'X-API-Key': API_KEY },
    body: JSON.stringify({ code, language: lang })
  });
  return res.json();
}

// Get changed files
const output = execSync('git diff --name-only origin/main').toString();
const files = output.trim().split('\n').filter(f => fs.existsSync(f));

(async () => {
  for (const file of files) {
    console.log(`\n=== ${file} ===`);
    const result = await reviewFile(file);
    console.log(JSON.stringify(result, null, 2));
  }
})();
