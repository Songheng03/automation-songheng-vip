My-automaton speaks MCP — the emerging standard for agent-to-service communication. Use any MCP-compatible agent to discover and call our services.

// Example: calling via MCP
fetch('https://automation.songheng.vip/api/mcp', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },

💻 Integration Code Examples

JavaScript / TypeScript

const API = 'https://automation.songheng.vip';

class AutomatonClient {
  constructor(apiKey) {
    this.apiKey = apiKey;
  }

  async analyze(text) {
    return this._call('/v1/analyze', { text, mode: 'analyze' });
  }

  async review(code) {
    return this._call('/v1/review', { code, language: 'auto' });
  }

  async scan(code) {
    return this._call('/v1/security', { code, language: 'auto' });
  }

  async _call(endpoint, body) {
    const res = await fetch(`${API}${endpoint}`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': this.apiKey
      },
      body: JSON.stringify(body)
    });
    if (res.status === 402) {
      throw new Error('Credits exhausted — purchase more at /pricing.html');
    }
    return res.json();
  }
}

// Usage
const client = new AutomatonClient('am_your_key_here');
const review = await client.review('function hello() { return "world"; }');
console.log(review);

Python

import requests

class AutomatonClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base = 'https://automation.songheng.vip'

    def analyze(self, text):
        r = requests.post(f'{self.base}/v1/analyze',
            json={'text': text, 'mode': 'analyze'},
            headers={'X-API-Key': self.api_key})
        if r.status_code == 402:
            raise Exception('Credits exhausted')
        return r.json()

    def review(self, code):
        r = requests.post(f'{self.base}/v1/review',
            json={'code': code, 'language': 'auto'},
            headers={'X-API-Key': self.api_key})
        return r.json()

client = AutomatonClient('am_your_key')
result = client.review('def hello(): return "world"')
print(result)
