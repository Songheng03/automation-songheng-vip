JavaScript (Node.js)

const API = 'https://automation.songheng.vip';

async function analyze(text) {
  const res = await fetch(`${API}/free/analyze`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text, mode: 'analyze' })
  });
  return res.json();
}

// With API key (paid)
async function review(code, lang) {
  const res = await fetch(`${API}/v1/review`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'am_YOUR_KEY_HERE'
    },
    body: JSON.stringify({ code, language: lang })
  });
  return res.json();
}

// Example usage
async function main() {
  const result = await analyze('AI agents are transforming...');
  console.log(result);
}

main();

Python

import requests

API = 'https://automation.songheng.vip'

def analyze_text(text):
    resp = requests.post(f'{API}/free/analyze', json={
        'text': text,
        'mode': 'analyze'
    })
    return resp.json()

def review_code(code, language='javascript'):
    resp = requests.post(f'{API}/free/review', json={
        'code': code,
        'language': language
    })
    return resp.json()

# Paid version with API key
def security_scan(code):
    resp = requests.post(f'{API}/v1/security', 
        json={'code': code},
        headers={'X-API-Key': 'am_YOUR_KEY_HERE'}
    )
    return resp.json()

if __name__ == '__main__':
    result = analyze_text('Python is a programming language...')
    print(result)
