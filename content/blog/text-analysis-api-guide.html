Integration Examples

// JavaScript
async function analyze(text) {
  const res = await fetch('https://automation.songheng.vip/free/analyze', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({text})
  });
  return res.json();
}
