The Setup

Add a single file to your repo: .github/workflows/ai-code-review.yml

Step 1: Add your API key as a secret

Go to your repo → Settings → Secrets and variables → Actions → New repository secret

Name: MY_AUTOMATON_KEY

Value: Your API key from the quick start above

Step 2: Create the workflow

Create .github/workflows/ai-code-review.yml with this content:

name: AI Code Review
on:
  pull_request:
    types: [opened, synchronize]
  pull_request_review:
    types: [requested]

jobs:
  code-review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: AI Code Review
        uses: fjogeleit/http-request-action@v1
        with:
          url: 'https://automation.songheng.vip/v1/review'
          method: 'POST'
          contentType: 'application/json'
          customHeaders: '{"X-API-Key": "${{ secrets.MY_AUTOMATON_KEY }}"}'
          data: >-
            {{
              toJSON({
                code: join(github.event.pull_request.diff_url),
                language: 'javascript',
                mode: 'full'
              })
            }}

      - name: Post Review Comment
        uses: actions/github-script@v7
        with:
          script: |
            const review = JSON.parse(${{ steps.code-review.outputs.response }});
            let comment = '## 🤖 AI Code Review\n\n';
            
            if (review.score) {
              comment += `**Overall Score:** ${review.score}/100\n\n`;
            }
            
            if (review.issues && review.issues.length > 0) {
              comment += '### Issues Found\n\n';
              review.issues.slice(0, 10).forEach(issue => {
                const severity = issue.severity || 'info';
                const emoji = severity === 'critical' ? '🔴' : 
                             severity === 'high' ? '🟠' : 
                             severity === 'medium' ? '🟡' : '⚪';
                comment += `${emoji} **${issue.message}**\n\n`;
                if (issue.line) comment += `   Line: ${issue.line}\n`;
                if (issue.suggestion) comment += `   Suggestion: ${issue.suggestion}\n`;
                comment += '\n';
              });
            } else {
              comment += '✅ No issues found!\n';
            }
            
            comment += '\n---\n*Powered by [my-automaton](https://automation.songheng.vip) AI*';
            
            await github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: comment
            });
