Quick Setup: GitHub Actions + Free API

Add this file to your repo at .github/workflows/ai-review.yml:

name: Free AI Code Review
on:
  pull_request:
    types: [opened, synchronize]

jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Get diff
        id: diff
        run: |
          git fetch origin ${{ github.base_ref }}
          git diff origin/${{ github.base_ref }} -- '*.py' '*.js' '*.ts' '*.go' '*.rs' '*.java' > /tmp/diff.txt
          echo "diff<> $GITHUB_OUTPUT
          cat /tmp/diff.txt >> $GITHUB_OUTPUT
          echo "EOF" >> $GITHUB_OUTPUT
      
      - name: AI Review
        id: review
        uses: fjogeleit/http-request-action@v1
        with:
          url: 'https://automation.songheng.vip/api/free/review'
          method: 'POST'
          contentType: 'application/json'
          data: '{"code":"${{ steps.diff.outputs.diff }}","language":"auto"}'
      
      - name: Post Comment
        uses: actions/github-script@v7
        with:
          script: |
            const review = JSON.parse('${{ steps.review.outputs.response }}');
            const body = review.error 
              ? `## 🤖 AI Review (Free)\n\n⚠️ ${review.error}\n\n_[Upgrade for unlimited reviews](${review.upgrade || "https://automation.songheng.vip/upgrade"})_`
              : `## 🤖 AI Code Review\n\n${review.explanation || 'Review complete.'}\n\n---\n_Free tier · 3 reviews/day left_`;
            github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.issue.number,
              body
            });
