Step 2: The API Gateway

Here's the core gateway that handles both free and paid requests:

const express = require('express');
const OpenAI = require('openai');
const x402 USDC = require('x402');
const { RateLimiterMemory } = require('rate-limiter-flexible');

const app = express();
const openai = new OpenAI({ apiKey: process.env.DEEPSEEK_KEY, baseURL: 'https://api.deepseek.com' });
const x402 = new x402 USDC(process.env.STRIPE_SK);

// Rate limiter for free tier: 3 requests/day per IP
const freeLimiter = new RateLimiterMemory({
  points: 3,
  duration: 86400, // 24 hours
  keyPrefix: 'free'
});

// --- AI Code Review Logic ---
const SYSTEM_PROMPT = \`You are an expert code reviewer. Analyze the provided code for:
1. Security vulnerabilities (OWASP Top 10)
2. Code quality issues (naming, complexity, duplication)
3. Anti-patterns and bugs
4. Performance issues
5. Style inconsistencies

Return a JSON object with:
- score: 0-100 overall quality score
- issues: array of {severity, type, message, line, remediation}
- summary: brief overview
\`;

async function reviewCode(code, language) {
  const response = await openai.chat.completions.create({
    model: 'deepseek-chat',
    messages: [
      { role: 'system', content: SYSTEM_PROMPT },
      { role: 'user', content: \`Review this \${language} code:\n\n\${code}\` }
    ],
    response_format: { type: 'json_object' },
    temperature: 0.3
  });
  return JSON.parse(response.choices[0].message.content);
}

// --- Free Endpoint ---
app.post('/free/review', async (req, res) => {
  try {
    const ip = req.ip;
    await freeLimiter.consume(ip); // 3/day limit

    const { code, language } = req.body;
    if (!code) return res.status(400).json({ error: 'Code required' });

    const result = await reviewCode(code, language || 'javascript');
    res.json(result);
  } catch (err) {
    if (err instanceof Error && err.name === 'RateLimiterError') {
      return res.status(429).json({ error: 'Free limit reached. Get API key for unlimited.' });
    }
    res.status(500).json({ error: 'Review failed' });
  }
});

// --- Paid Endpoint ---
app.post('/v1/review', async (req, res) => {
  const apiKey = req.headers['x-api-key'];
  const user = API_KEYS[apiKey];
  if (!user || user.credits < 5) {
    return res.status(402).json({ error: 'Insufficient credits', price: '5 credits' });
  }

  const { code, language } = req.body;
  const result = await reviewCode(code, language || 'javascript');

  user.credits -= 5;
  res.json({ ...result, creditsRemaining: user.credits });
});

Step 4: Deploy with Docker

# Dockerfile
FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm ci --production
COPY . .
EXPOSE 8080
CMD ["node", "gateway.js"]
# docker-compose.yml
version: '3.8'
services:
  gateway:
    build: .
    ports:
      - "8080:8080"
    environment:
      - DEEPSEEK_KEY=\${DEEPSEEK_KEY}
      - STRIPE_SK=\${STRIPE_SK}
    restart: unless-stopped
