What is x402?

x402 (named after the HTTP 402 "Payment Required" status code) is a protocol for pay-per-request API access using cryptocurrency. Here's how it works:

  1. User calls your API endpoint
  2. Your server responds with HTTP 402 + payment instructions (price, wallet, chain)
  3. User sends the exact amount in USDC to your wallet on Base chain
  4. User retries the request with the transaction hash as proof
  5. Your server verifies the payment and processes the request

This is perfect for AI services where each inference costs you money — you only serve requests that are paid for.

Step 2: Create the Gateway

Your gateway is the entry point for all API requests. Here's a minimal Express-based gateway:

// gateway.js - Minimal AI API Gateway
import express from 'express';
const app = express();
app.use(express.json());

const PORT = 8080;
const WALLET = '0x76eADdEBFfb6A61DD071f97F4508467fc55dd113';
const CHAIN = 'base';

// Pricing table
const PRICING = {
  'analyze': { cost: '0.01', description: 'Deep text analysis' },
  'summarize': { cost: '0.02', description: 'AI summarization' },
  'review': { cost: '0.05', description: 'Full code review' },
};

// x402 handler middleware
function x402Middleware(service) {
  return async (req, res) => {
    const price = PRICING[service];
    
    // Check for payment
    const txHash = req.headers['x-x402-payment'];
    
    if (!txHash) {
      // Return 402 with payment instructions
      return res.status(402).json({
        error: 'payment_required',
        payment: {
          amount: price.cost,
          currency: 'USDC',
          chain: CHAIN,
          wallet: WALLET,
          service: service
        }
      });
    }
    
    // Verify payment (simplified — in production, check tx on-chain)
    // ... verification logic ...
    
    // Process the request
    const result = await callAI(req.body);
    res.json(result);
  };
}

// Register endpoints
app.post('/v1/analyze', x402Middleware('analyze'));
app.post('/v1/summarize', x402Middleware('summarize'));
app.post('/v1/review', x402Middleware('review'));

app.listen(PORT, () => console.log(`Gateway on :${PORT}`));

Step 4: Integrate AI Inference

Use any AI provider — OpenAI, Anthropic, DeepSeek, or run local models:

// inference.js — AI service layer
const DEEPSEEK_API = 'https://api.deepseek.com/v1/chat/completions';
const API_KEY = process.env.DEEPSEEK_API_KEY;

async function callAI({ text, mode }) {
  const systemPrompts = {
    analyze: 'Analyze this text in detail...',
    summarize: 'Summarize this text concisely...',
    review: 'Review this code for issues...'
  };

  const response = await fetch(DEEPSEEK_API, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'deepseek-chat',
      messages: [
        { role: 'system', content: systemPrompts[mode] },
        { role: 'user', content: text }
      ],
      max_tokens: 1000
    }
  });

  const data = await response.json();
  return { result: data.choices[0].message.content };
}

// free-tier.js — Usage tracking
const freeUsage = new Map(); // ip -> { date, count }

function checkFreeTier(ip) {
  const today = new Date().toISOString().split('T')[0];
  const record = freeUsage.get(ip);
  
  if (!record || record.date !== today) {
    freeUsage.set(ip, { date: today, count: 0 });
    return { allowed: true, remaining: 2 };
  }
  
  if (record.count >= 3) {
    return { allowed: false, remaining: 0 };
  }
  
  record.count++;
  return { allowed: true, remaining: 3 - record.count };
}
