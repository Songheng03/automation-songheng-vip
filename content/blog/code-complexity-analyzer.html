Code Complexity Analyzer — Cyclomatic & Cognitive Complexity Guide

Why Complexity Analysis Matters

High-complexity code is the #1 predictor of bugs, maintenance costs, and developer turnover. Research shows that functions with cyclomatic complexity > 10 have 3x more defects. Functions with complexity > 20 have 10x more defects.

Understanding the Metrics

Cyclomatic Complexity

Measures the number of linearly independent paths through your code. Each if, while, for, case, &&, ||, catch adds 1 to the count.

  • 1-10: Simple, low risk
  • 11-20: Moderate, some risk
  • 21-50: Complex, high risk — should refactor
  • 50+: Untestable — must refactor

Cognitive Complexity

A modern metric that measures how hard code is to understand. Unlike cyclomatic complexity, it penalizes nesting and breaks — not just branching.

  • 0-5: Easy to understand
  • 6-15: Moderate effort
  • 16-30: Difficult — should refactor
  • 30+: Very difficult — must refactor

Example: Refactoring High-Complexity Code

Before (Complexity: 15 cyclomatic, 24 cognitive)

function processOrder(order, user, inventory) {
  let total = 0;
  let errors = [];
  
  if (order && order.items) {
    for (let i = 0; i < order.items.length; i++) {
      const item = order.items[i];
      if (item && item.id && inventory[item.id]) {
        const inv = inventory[item.id];
        if (inv.quantity >= item.quantity) {
          if (user.tier === 'premium') {
            total += inv.price * item.quantity * 0.9;
          } else if (user.tier === 'enterprise') {
            total += inv.price * item.quantity * 0.8;
          } else {
            total += inv.price * item.quantity;
          }
        } else {
          errors.push({ item: item.id, error: 'Insufficient stock' });
        }
      } else {
        errors.push({ item: item?.id || 'unknown', error: 'Item not found' });
      }
    }
  }
  
  return { total, errors };
}

After (Complexity: 2 cyclomatic, 3 cognitive)

function getDiscount(user) {
  if (user.tier === 'premium') return 0.9;
  if (user.tier === 'enterprise') return 0.8;
  return 1.0;
}

function processItem(item, inventory, discount) {
  if (!item?.id || !inventory[item.id]) {
    return { error: 'Item not found', item: item?.id || 'unknown' };
  }
  
  const inv = inventory[item.id];
  if (inv.quantity < item.quantity) {
    return { error: 'Insufficient stock', item: item.id };
  }
  
  return { price: inv.price * item.quantity * discount };
}

function processOrder(order, user, inventory) {
  if (!order?.items) return { total: 0, errors: [] };
  
  const discount = getDiscount(user);
  const results = order.items.map(item => processItem(item, inventory, discount));
  
  return {
    total: results.reduce((sum, r) => sum + (r.price || 0), 0),
    errors: results.filter(r => r.error)
  };
}
