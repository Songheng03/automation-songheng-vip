// Full client library — copy and use
const BASE = 'https://automation.songheng.vip';
const WALLET = '0x76eADdEBFfb6A61DD071f97F4508467fc55dd113';

async function callAI(endpoint, data) {
  // First try with free tier
  let res = await fetch(`${BASE}${endpoint}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });

  // If payment required (HTTP 402), pay with USDC
  if (res.status === 402) {
    const txHash = await sendUSDC(WALLET, getCost(endpoint));
    res = await fetch(`${BASE}${endpoint}`, {
      method: 'POST',
      headers: { 
        'Content-Type': 'application/json',
        'X-X402-Payment': txHash
      },
      body: JSON.stringify(data)
    });
  }

  return res.json();
}
