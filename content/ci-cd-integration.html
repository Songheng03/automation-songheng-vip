Create .github/workflows/ai-code-review.yml in your repo:

name: AI Code Review
on:
  pull_request:
    types: [opened, synchronize]

jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Get changed files
        id: files
        uses: jitterbit/get-changed-files@v1
      - name: AI Code Review
        run: |
          for file in ${{ steps.files.outputs.all }}; do
            if [[ "$file" =~ \.(js|ts|py|go|rs|java|cpp|rb|php)$ ]]; then
              echo "=== Reviewing $file ==="
              curl -s -X POST https://automation.songheng.vip/free/review \
                -H "Content-Type: application/json" \
                -d "{\"text\":\"$(cat $file | head -500)\",\"language\":\"${file##*.}\"}"
              echo ""
            fi
          done

// Jenkinsfile
pipeline {
  agent any
  stages {
    stage('AI Code Review') {
      steps {
        script {
          def files = sh(script: 'git diff --name-only origin/main...HEAD', returnStdout: true).trim().split('\n')
          files.each { file ->
            if (file =~ /\\.(js|ts|py|go|rs|java|cpp|rb|php)$/) {
              echo "Reviewing: ${file}"
              def code = readFile(file).take(10000)
              sh """
                curl -s -X POST https://automation.songheng.vip/free/review \
                  -H "Content-Type: application/json" \
                  -d '{"text":"${code.replace(/'/,"\\\\'")}","language":"${file.split('\\.').last()}"}'
              """
            }
          }
        }
      }
    }
  }
}
