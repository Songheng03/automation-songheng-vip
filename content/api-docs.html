JavaScript / Node.js

const API = "https://automation.songheng.vip";

// Free tier

  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ text: "Long text to summarize..." })
}).then(r => r.json()).then(d => console.log(d.result));

// Premium

  method: "POST",

  body: JSON.stringify({ text: "complexFunction(a => a.map(b => b.c))" })
}).then(r => r.json()).then(d => console.log(d.result));

GitHub CI/CD Integration

# .github/workflows/ai-review.yml
name: AI Code Review
on: [pull_request]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: AI Code Review
        run: |
          DIFF=$(git diff origin/main -- '*.js' '*.ts')
          curl -X POST https://automation.songheng.vip/v1/review \
            -H "Content-Type: application/json" \
            -H "X-API-Key: ${{ secrets.AI_API_KEY }}" \
            -d "{\"text\": $(echo "$DIFF" | jq -Rs .)}"
