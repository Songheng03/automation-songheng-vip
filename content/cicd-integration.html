GitHub Actions PR Review Workflow

Creates .github/workflows/automaton-review.yml

Copy
name: AI Code Review
on:
  pull_request:
    types: [opened, synchronize]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run automaton review
        id: review
        run: |
          DIFF=$(git diff origin/${{ github.base_ref }}...HEAD -- . | head -2000)
          curl -s -X POST https://automation.songheng.vip/v1/review \
            -H "Content-Type: application/json" \
            -H "X-API-Key: ${{ secrets.AUTOMATON_API_KEY }}" \
            -d "{\"code\":$(echo "$DIFF" | jq -Rs .),\"language\":\"diff\"}" \
            > review.json
          cat review.json | jq -r '.result // .review' > review.md
      - name: Post comment
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            const review = fs.readFileSync('review.md', 'utf8');
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: `## 🤖 AI Code Review\n\n${review.slice(0, 60000)}\n\n---\n*Reviewed by [my-automaton](https://automation.songheng.vip)*`
            });

GitLab CI Merge Request Review

Copy
# .gitlab-ci.yml
automaton-review:
  image: curlimages/curl:latest
  only:
    - merge_requests
  script:
    - |
      DIFF=$(curl -s "$CI_API_V4_URL/projects/$CI_PROJECT_ID/merge_requests/$CI_MERGE_REQUEST_IID/diffs" | jq -r '.[].diff' | head -2000)
      curl -s -X POST https://automation.songheng.vip/v1/review \
        -H "Content-Type: application/json" \
        -H "X-API-Key: $AUTOMATON_API_KEY" \
        -d "{\"code\":$(echo "$DIFF" | jq -Rs .),\"language\":\"diff\"}" \
        | jq -r '.result // .review'
  variables:
    GIT_STRATEGY: none

Node.js Script for Any CI

For Jenkins, CircleCI, or custom pipelines:

Copy
const https = require('https');
const fs = require('fs');

async function reviewDiff() {
  const diff = fs.readFileSync(process.env.DIFF_FILE, 'utf8');
  const res = await fetch('https://automation.songheng.vip/v1/review', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': process.env.AUTOMATON_API_KEY
    },
    body: JSON.stringify({ code: diff, language: 'diff' })
  });
  const data = await res.json();
  console.log(data.result || data.review);
}
reviewDiff().catch(console.error);
