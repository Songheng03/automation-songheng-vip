async function callAPI(endpoint, data) {
  const url = `https://automation.songheng.vip${endpoint}`;
  let res = await fetch(url, {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify(data)
  });
  
  if (res.status === 402) {
    const info = await res.json();
    console.log('Send', info.payment.amount_cents + '¢', 'to', info.payment.wallet);
    const txHash = await window.ethereum.request({
      method: 'eth_sendTransaction',
      params: [{ to: info.payment.wallet, value: '0x...' }]
    });
    res = await fetch(url, {
      method: 'POST',
      headers: {'Content-Type': 'application/json', 'X-X402-Payment': txHash},
      body: JSON.stringify(data)
    });
  }
  return res.json();
}
