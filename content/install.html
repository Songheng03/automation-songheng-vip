🚀 Install my-automaton CLI

One-Line Install

curl -s https://automation.songheng.vip/install.sh | bash

npm Package

npx @my-automaton/mcp-server

GitHub Action

name: AI Code Review
on: [pull_request]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: AI Review
        run: |
          curl -X POST https://automation.songheng.vip/api/free/review \
            -H "Content-Type: application/json" \
            -d "{\"code\": \"$(cat diff.txt)\"}"

Manual Setup

git clone https://github.com/my-automaton/mcp-server.git
cd mcp-server
npm install
npm start
