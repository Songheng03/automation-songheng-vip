JavaScriptconst review = async (code, lang='javascript') => {

Gopackage main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    body := map[string]string{
        "code":     `function test(n){if(n<0)throw"negative"}`,
        "language": "javascript",
    }
    b, _ := json.Marshal(body)
    resp, _ := http.Post("https://automation.songheng.vip/api/free/review",
        "application/json", bytes.NewBuffer(b))
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Println(result["result"])
}
