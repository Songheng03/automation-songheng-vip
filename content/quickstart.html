const https = require('https');

const data = JSON.stringify({
  text: 'function hello() { return 42; }',
  mode: 'analyze'
});

const options = {
  hostname: 'automation.songheng.vip',
  port: 443,
  path: '/v1/analyze',
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-API-Key': 'YOUR_API_KEY_HERE',
    'Content-Length': Buffer.byteLength(data)
  }
};

const req = https.request(options, (res) => {
  let body = '';
  res.on('data', (chunk) => body += chunk);
  res.on('end', () => {
    const result = JSON.parse(body);
    console.log('Analysis:', result);
  });
});

req.write(data);
req.end();
import requests
import json

url = 'https://automation.songheng.vip/v1/analyze'
headers = {
    'Content-Type': 'application/json',
    'X-API-Key': 'YOUR_API_KEY_HERE'
}
data = {
    'text': 'function hello() { return 42; }',
    'mode': 'analyze'
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
print('Analysis:', result)
curl -X POST https://automation.songheng.vip/v1/analyze \
  -H "Content-Type: application/json" \
  -H "X-API-Key: YOUR_API_KEY_HERE" \
  -d '{
    "text": "function hello() { return 42; }",
    "mode": "analyze"
  }'
⚡ Replace: Change YOUR_API_KEY_HERE to your actual API key (starts with am_).

Need Help?

If you're having trouble, here are common issues:

  • 401 Unauthorized: Check that your API key is correct and hasn't expired
  • 402 Payment Required: You've used all your credits. Purchase more at /pricing
  • 429 Too Many Requests: You've hit the rate limit (100 requests/minute). Wait a moment and try again
