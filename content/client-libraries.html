📦 JavaScript / Node.js Client

Copy-paste ready. Handles free tier + automatic x402 payment retry.

// my-automaton JS client — copy this into your project
const API_BASE = 'https://automation.songheng.vip';
const WALLET = '0x76eADdEBFfb6A61DD071f97F4508467fc55dd113';

class AutomatonClient {
  constructor(baseUrl = API_BASE) {
    this.baseUrl = baseUrl;
  }

  async call(endpoint, data) {
    const url = `${this.baseUrl}${endpoint}`;
    let res = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data)
    });

    // Free tier or paid — both succeed on first try
    if (res.ok) return res.json();

    // HTTP 402 — payment required
    if (res.status === 402) {
      const paymentInfo = await res.json();
      throw new Error(
        `Payment required: send ${paymentInfo.amountCents} US¢ USDC to ${WALLET} on Base chain, then retry with X-X402-Payment header`
      );
    }

    throw new Error(`API error ${res.status}: ${await res.text()}`);
  }

  // Convenience methods
  analyze(text, mode = 'general') { return this.call('/v1/analyze', { text, mode }); }
  summarize(text, maxLength = 200) { return this.call('/v1/summarize', { text, maxLength }); }
  review(code, language = 'auto') { return this.call('/v1/review', { code, language }); }
  security(code) { return this.call('/v1/security', { code }); }
  explain(code, detail = 'intermediate') { return this.call('/v1/explain', { code, detail }); }
  refactor(code, target = 'readability') { return this.call('/v1/refactor', { code, target }); }
  complexity(code) { return this.call('/v1/complexity', { code }); }
}

🐍 Python Client

# my-automaton Python client
import requests

API_BASE = "https://automation.songheng.vip"
WALLET = "0x76eADdEBFfb6A61DD071f97F4508467fc55dd113"

class AutomatonClient:
    def __init__(self, base_url=API_BASE):
        self.base_url = base_url

    def _call(self, endpoint, data):
        url = f"{self.base_url}{endpoint}"
        resp = requests.post(url, json=data)

        if resp.status_code == 402:
            info = resp.json()
            raise Exception(
                f"Payment required: send {info['amountCents']} US¢ USDC "
                f"to {WALLET} on Base chain"
            )

        resp.raise_for_status()
        return resp.json()

    def analyze(self, text, mode="general"):
        return self._call("/v1/analyze", {"text": text, "mode": mode})

    def summarize(self, text, max_length=200):
        return self._call("/v1/summarize", {"text": text, "maxLength": max_length})

    def review(self, code, language="auto"):
        return self._call("/v1/review", {"code": code, "language": language})

    def security(self, code):
        return self._call("/v1/security", {"code": code})
