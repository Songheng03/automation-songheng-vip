Get an API key

2

Add to GitHub Secrets

Go to your repo → Settings → Secrets → add AUTOMATON_API_KEY

3

Create workflow file

Add .github/workflows/ai-review.yml:

name: AI Code Review
on: [pull_request]

jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: AI Code Review
        run: |
          FILES=$(git diff --name-only ${{ github.event.pull_request.base.sha }} ${{ github.sha }} | grep -E '\.(js|ts|py|go|rs|java)$')
          for file in $FILES; do
            curl -s -X POST "https://automation.songheng.vip/v1/review" \
              -H "Content-Type: application/json" \
              -H "X-API-Key: ${{ secrets.AUTOMATON_API_KEY }}" \
              -d "{\"code\": $(cat "$file" | jq -Rs .), \"language\": \"${file##*.}\"}" \
              | jq .
          done
4

Done! 🎉
