CI/CD Integration Script

#!/usr/bin/env python3
"""Auto-review changed files in a PR"""
import sys, os, requests, subprocess, json

API_KEY = os.environ.get("MY_AUTOMATON_KEY")
if not API_KEY:
    print("❌ Set MY_AUTOMATON_KEY env var")
    sys.exit(1)

BASE_URL = "https://automation.songheng.vip"

# Get changed files from git diff
result = subprocess.run(
    ["git", "diff", "--name-only", "origin/main...HEAD"],
    capture_output=True, text=True
)
files = [f for f in result.stdout.split("\n") if f.endswith((".py", ".js", ".ts", ".go"))]

for filepath in files[:5]:  # limit to 5 files
    with open(filepath) as f:
        code = f.read()
    resp = requests.post(
        f"{BASE_URL}/v1/review",
        headers={"Content-Type": "application/json", "X-API-Key": API_KEY},
        json={"code": code, "language": filepath.split(".")[-1]}
    )
    review = resp.json()
    print(f"\n## 📝 {filepath}")
    print(review.get("review", review.get("analysis", json.dumps(review, indent=2))))

🔵 Go

package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

const (
    baseURL = "https://automation.songheng.vip"
    apiKey  = "am_your_key_here"
)

type AnalyzeRequest struct {
    Text string `json:"text"`
    Mode string `json:"mode"`
}

func analyzeText(text, mode string) {
    body, _ := json.Marshal(AnalyzeRequest{Text: text, Mode: mode})
    req, _ := http.NewRequest("POST", baseURL+"/v1/analyze", bytes.NewBuffer(body))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("X-API-Key", apiKey)
    
    resp, _ := http.DefaultClient.Do(req)
    defer resp.Body.Close()
    
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Printf("%+v\n", result)
}

func main() {
    analyzeText("Go is a statically typed, compiled language...", "analyze")
}

🦀 Rust

// Add to Cargo.toml:
// [dependencies]
// reqwest = { version = "0.12", features = ["json"] }
// serde = { version = "1.0", features = ["derive"] }
// tokio = { version = "1", features = ["full"] }

use serde::{Serialize, Deserialize};
use reqwest::Client;

#[derive(Serialize)]
struct ReviewRequest {
    code: String,
    language: String,
}

#[derive(Deserialize, Debug)]
struct ReviewResponse {
    review: Option<String>,
    issues: Option<Vec<String>>,
    error: Option<String>,
}

#[tokio::main]
async fn main() -> Result<(), Box> {
    let client = Client::new();
    let req = ReviewRequest {
        code: "fn add(x: i32, y: i32) -> i32 { x + y }".to_string(),
        language: "rust".to_string(),
    };
    
    let resp = client.post("https://automation.songheng.vip/v1/review")
        .header("X-API-Key", "am_your_key_here")
        .json(&req)
        .send()
        .await?;
    
    let result: ReviewResponse = resp.json().await?;
    println!("{:#?}", result);
    Ok(())
}

🔄 CI/CD Integration (GitHub Actions)

# .github/workflows/ai-code-review.yml
name: AI Code Review
on: [pull_request]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: AI Review Changed Files
        env:
          MY_AUTOMATON_KEY: \${{ secrets.MY_AUTOMATON_KEY }}
        run: |
          CHANGED_FILES=$(git diff --name-only origin/main...HEAD | grep -E '\.(py|js|ts|go|rs)$' || true)
          for file in $CHANGED_FILES; do
            echo "::group::Reviewing $file"
            curl -s https://automation.songheng.vip/v1/review \
              -H "Content-Type: application/json" \
              -H "X-API-Key: $MY_AUTOMATON_KEY" \
              -d "{\"code\": $(jq -Rs . < $file), \"language\": \"${file##*.}\"}" \
              | jq .
            echo "::endgroup::"
          done
